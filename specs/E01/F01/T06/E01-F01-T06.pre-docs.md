## 1. Executive Summary

### What We're Building

A per-project configuration system for power/ground net classification with:
1. **YAML Config File**: `{project}/.ink/net_classification.yaml`
2. **Qt Configuration Dialog**: Edit net names and patterns visually
3. **NetNormalizer Integration**: Factory method to create normalizers from config
4. **Menu Integration**: Edit > Net Classification...

### Why This Matters

- Different PDKs use different power/ground naming conventions
- Users shouldn't need to modify code to add custom net names
- Per-project config allows different settings for different designs
- Visual editor makes configuration accessible to non-programmers

### Phase 1: Configuration Infrastructure

**Goal**: YAML-based configuration loading and saving

#### Step 1.1: Create Config Dataclass

```python
# src/ink/infrastructure/config/net_classification_config.py

from dataclasses import dataclass, field
from pathlib import Path

@dataclass
class NetClassificationConfig:
    """Configuration for power/ground net classification.

    Attributes:
        power_names: Exact power net names (case-insensitive matching)
        power_patterns: Regex patterns for power nets
        ground_names: Exact ground net names (case-insensitive matching)
        ground_patterns: Regex patterns for ground nets
        override_defaults: If True, ignore default VDD/VSS patterns
    """
    power_names: list[str] = field(default_factory=list)
    power_patterns: list[str] = field(default_factory=list)
    ground_names: list[str] = field(default_factory=list)
    ground_patterns: list[str] = field(default_factory=list)
    override_defaults: bool = False
```

#### Step 1.2: Add YAML Load/Save Methods

```python
import yaml
from pathlib import Path

CONFIG_FILENAME = "net_classification.yaml"
CONFIG_DIR = ".ink"

@classmethod
def load(cls, project_path: Path) -> "NetClassificationConfig":
    """Load configuration from project directory."""
    config_file = project_path / CONFIG_DIR / CONFIG_FILENAME
    if not config_file.exists():
        return cls()  # Return empty defaults

    with open(config_file, encoding="utf-8") as f:
        data = yaml.safe_load(f) or {}

    return cls(
        power_names=data.get("power_nets", {}).get("names", []),
        power_patterns=data.get("power_nets", {}).get("patterns", []),
        ground_names=data.get("ground_nets", {}).get("names", []),
        ground_patterns=data.get("ground_nets", {}).get("patterns", []),
        override_defaults=data.get("override_defaults", False),
    )

def save(self, project_path: Path) -> None:
    """Save configuration to project directory."""
    config_dir = project_path / CONFIG_DIR
    config_dir.mkdir(parents=True, exist_ok=True)

    data = {
        "version": 1,
        "power_nets": {
            "names": self.power_names,
            "patterns": self.power_patterns,
        },
        "ground_nets": {
            "names": self.ground_names,
            "patterns": self.ground_patterns,
        },
        "override_defaults": self.override_defaults,
    }

    config_file = config_dir / CONFIG_FILENAME
    with open(config_file, "w", encoding="utf-8") as f:
        yaml.dump(data, f, default_flow_style=False, sort_keys=False)
```

#### Step 2.2: Add Pattern Management Methods

```python
def add_power_patterns(self, patterns: Iterable[str]) -> None:
    """Add custom power net patterns."""
    self._custom_power_patterns.update(patterns)
    self._net_cache.clear()  # Invalidate cache

def add_ground_patterns(self, patterns: Iterable[str]) -> None:
    """Add custom ground net patterns."""
    self._custom_ground_patterns.update(patterns)
    self._net_cache.clear()  # Invalidate cache

def clear_default_patterns(self) -> None:
    """Disable default patterns (use only custom)."""
    self._use_default_patterns = False
    self._net_cache.clear()
```

#### Step 2.3: Factory Method

```python
@classmethod
def from_config(cls, config: NetClassificationConfig) -> "NetNormalizer":
    """Create NetNormalizer from configuration."""
    normalizer = cls(
        power_nets=config.power_names,
        ground_nets=config.ground_names,
    )

    if config.power_patterns:
        normalizer.add_power_patterns(config.power_patterns)
    if config.ground_patterns:
        normalizer.add_ground_patterns(config.ground_patterns)

    if config.override_defaults:
        normalizer.clear_default_patterns()

    return normalizer
```

```python
def _create_net_tab(self, net_type: Literal["power", "ground"]) -> QWidget:
    """Create a tab for power or ground configuration."""
    widget = QWidget()
    layout = QVBoxLayout(widget)

    # Names section
    names_group = self._create_list_section(
        title="Net Names (exact match, case-insensitive)",
        items=getattr(self._config, f"{net_type}_names"),
        list_name=f"{net_type}_names"
    )
    layout.addWidget(names_group)

    # Patterns section
    patterns_group = self._create_list_section(
        title="Regex Patterns (additional to defaults)",
        items=getattr(self._config, f"{net_type}_patterns"),
        list_name=f"{net_type}_patterns"
    )
    layout.addWidget(patterns_group)

    return widget
```

#### Step 3.3: Add/Remove Functionality

```python
def _on_add_item(self, list_widget: QListWidget) -> None:
    """Add new item to list via input dialog."""
    text, ok = QInputDialog.getText(
        self,
        "Add Item",
        "Enter net name or pattern:"
    )
    if ok and text.strip():
        list_widget.addItem(text.strip())

def _on_remove_item(self, list_widget: QListWidget) -> None:
    """Remove selected item from list."""
    current = list_widget.currentRow()
    if current >= 0:
        list_widget.takeItem(current)
```

```python
def _open_net_classification_dialog(self) -> None:
    """Open the net classification configuration dialog."""
    # Get current project path (if any)
    project_path = self._get_current_project_path()
    if not project_path:
        QMessageBox.warning(
            self,
            "No Project",
            "Please open a netlist file first."
        )
        return

    # Load current config
    config = NetClassificationConfig.load(project_path)

    # Show dialog
    dialog = NetClassificationDialog(config, self)
    if dialog.exec() == QDialog.Accepted:
        new_config = dialog.get_config()
        new_config.save(project_path)
        # Reload normalizer with new config
        self._reload_net_normalizer(new_config)
```
